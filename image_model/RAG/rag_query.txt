import torch
import faiss
import pickle
import numpy as np
from PIL import Image
from torchvision import transforms
from torchvision.models import resnet18

DEVICE = "cuda" if torch.cuda.is_available() else "cpu"
CHECKPOINT = "oct_resnet18.pth"
INDEX_PATH = "oct_faiss.index"
META_PATH = "oct_metadata.pkl"

transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406],
                         [0.229, 0.224, 0.225])
])

checkpoint = torch.load(CHECKPOINT, map_location=DEVICE)
class_names = checkpoint["class_names"]

classifier = resnet18(weights=None)
classifier.fc = torch.nn.Linear(classifier.fc.in_features, len(class_names))
classifier.load_state_dict(checkpoint["model_state_dict"])
classifier.to(DEVICE)
classifier.eval()

embedder = resnet18(weights=None)
embedder.fc = torch.nn.Linear(embedder.fc.in_features, len(class_names))
embedder.load_state_dict(checkpoint["model_state_dict"])
embedder.fc = torch.nn.Identity()
embedder.to(DEVICE)
embedder.eval()

index = faiss.read_index(INDEX_PATH)

with open(META_PATH, "rb") as f:
    metadata = pickle.load(f)


def query_image(image_path, top_k=5):
    img = Image.open(image_path).convert("RGB")
    x = transform(img).unsqueeze(0).to(DEVICE)

    with torch.no_grad():
        logits = classifier(x)
        probs = torch.softmax(logits, dim=1)[0]
        pred_idx = probs.argmax().item()

        emb = embedder(x).cpu().numpy()[0]

    emb = emb / np.linalg.norm(emb)
    emb = emb.astype("float32").reshape(1, -1)

    scores, indices = index.search(emb, top_k)

    results = []
    for score, idx in zip(scores[0], indices[0]):
        results.append({
            "similarity": float(score),
            "image_path": metadata[idx]["path"],
            "label": metadata[idx]["label"]
        })

    return {
        "prediction": class_names[pred_idx],
        "confidence": float(probs[pred_idx]),
        "similar_images": results
    }


if __name__ == "__main__":
    result = query_image("test_image.jpeg", top_k=5)

    print("Prediction:", result["prediction"])
    print("Confidence:", result["confidence"])

    print("\nSimilar Images:")
    for r in result["similar_images"]:
        print(r)
